loads.resstock

RESstock data accessor

The RESstock class is a Pandas data frame loaded with the RESstock building data.

Residental building types are coded using three characters R for residential, {'SA','SD','SM','LM','MH'} for single-family attached, single-family detached, small multi-family, large multi-family, and mobile-home. Values are given in W/unit.

Example:

To get the detached single-family home load data for Alameda CA use the command

from loads.resstock import RESstock
print(RESstock(state="CA",county="Alameda",building_type="RSD"))

133    def __init__(self,
134        state:str,
135        county:str=None,
136        building_type:list[str]=None,
137        freq:str|None="1h",
138        ):
139        """Construct a RESstock data frame
140
141        Arguments:
142
143        - `state`: specifies the state (e.g., "CA")
144
145        - `county`: specifies the county (e.g., "Alameda") or None for the
146          entire state
147
148        - `building_type`: specifies the building type (e.g., "house")
149
150        - `freq`: specifies the sampling interval (None for raw sampling)
151        """
152        assert building_type in self.BUILDING_TYPES, \
153            f"{building_type=} is not one of {self.BUILDING_TYPES}"
154
155        if self.CACHEDIR is None:
156            self.CACHEDIR = os.path.join(os.path.dirname(__file__),".cache")
157        os.makedirs(self.CACHEDIR,exist_ok=True)
158
159        # check cache
160        file = f"{state}_{building_type}.csv.gz" \
161            if county is None \
162            else f"{state}_{county}_{building_type}.csv.gz"
163        cache = os.path.join(self.CACHEDIR,file.replace(" ","-"))
164        if not os.path.exists(cache):
165
166            # download data to cache
167            root = "https://oedi-data-lake.s3.amazonaws.com/nrel-pds-building-stock/" \
168                "end-use-load-profiles-for-us-building-stock/2021/"\
169                "resstock_amy2018_release_1/timeseries_aggregates"
170            btype = self.BUILDING_TYPES[building_type]
171            if county is None:
172                url = f"{root}/by_state/state={state.upper()}/{state.lower()}-{btype}.csv"
173            else:
174                fips = County(ST=state,COUNTY=county).FIPS
175                url = f"{root}/by_county/state={state.upper()}/g{fips[:2]}0{fips[2:]}0-{btype}.csv"
176            try:
177                data = pd.read_csv(url)
178            except urllib.error.HTTPError as err:
179
180                # download error (most likely no data in RESstock)
181                warnings.warn(f"RESstock building type '{btype}' has no data ({err})")
182
183                # create all zeros dataframe
184                ndx = pd.date_range(
185                    start="2018-01-01 05:00:00+00:00",
186                    end="2019-01-01 04:00:00+00:00",
187                    freq=freq)
188                zeros = [0.0]*len(ndx)
189                data = pd.DataFrame(data={x:zeros for x in self.COLUMNS},index=ndx)
190                data.index.name = "timestamp"
191                data.reset_index(inplace=True)
192                data["units_represented"] = 0.0
193
194            data.to_csv(cache,compression="gzip" if cache.endswith(".gz") else None)
195
196        # load data from cache
197        data = pd.read_csv(cache,dtype=str,na_filter=False,low_memory=False)
198        data.set_index(["timestamp"],inplace=True)
199        data.index = (pd.DatetimeIndex(data.index,tz=pytz.timezone("EST")) \
200            - dt.timedelta(minutes=15)).tz_convert(pytz.UTC)
201
202        # capture number of housing units
203        units = data["units_represented"].astype(float)
204        if units.min() != units.max():
205            warnings.warn(f"{state=} {county=} number of units changes (using max)")
206        units = units.max()
207
208        # restructure data
209        data.drop([x for x in data.columns if x not in self.COLUMNS],inplace=True,axis=1)
210        data.rename(self.COLUMNS,inplace=True,axis=1)
211        for value in self.COLUMNS.values():
212            data[value] = [_float(x)/units*1000 for x in data[value]]
213
214        data["units"] = units
215
216        # resample if necessary
217        if not freq is None:
218            ts = data.index.diff().mean().total_seconds()/3600
219            data = (data/ts).resample(freq).ffill()
220
221        # move year-end data to beginning
222        data.index = pd.DatetimeIndex([str(x).replace("2019","2018") for x in data.index])
223        super().__init__(data.sort_index())
